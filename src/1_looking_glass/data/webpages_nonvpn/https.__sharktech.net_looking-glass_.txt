<script>/**
 * Makes API call for looking glass
 * ping or traceroute
 */
function pingTrace(e) {
    var need = document.getElementById('host').value;

    if (document.getElementById('util').value === "ping") {
        document.getElementById('changing').innerHTML = `<p class="line1" id = "changing"><t style = "color: white;">[user@sharktech ~ ] %</t> ping `+need +`<span class="cursor1">_</span></p>`;
    } else {
        document.getElementById('changing').innerHTML = `<p class="line1" id = "changing"><t style = "color: white;">[user@sharktech ~ ] %</t> trace `+need +`<span class="cursor1">_</span></p>`;

    }
    //  e.preventDefault();

    let request = null;
    let server_url = null;

    // Document elements
    let dc_element = document.getElementById('datacenter');
    let tool_element = document.getElementById('util');
    let results_element = document.getElementById('lgResultScreen');
    let host_element = document.getElementById('host');
    let submit_element = document.getElementById('toolSubmit');

    // Build URL
    let datacenter = dc_element.options[dc_element.selectedIndex].value;
    let tool = tool_element.options[tool_element.selectedIndex].value;
    let host = host_element.value;

    switch(datacenter) {
        case 'ams':
            server_url = `https://ams.sharktech.net:8000/${tool}/${host}`
            break;
        case "chi":
            server_url = `https://chi.sharktech.net:8000/${tool}/${host}`
            break;
        case "den":
            server_url = `https://den.sharktech.net:8000/${tool}/${host}`
            break;
        case "lax":
            server_url = `https://la.sharktech.net:8000/${tool}/${host}`
    }

    // Check host
    if(!host) {
        results_element.style.display = "block";
        results_element.innerHTML = `<p class="line2" style="color: red;">*You must specify a host!</p><br/><br/>`;
    } else {
        // Disable submit button
        submit_element.disabled = true;
        let requestData = JSON.stringify({
            host: host,
            datacenter: datacenter,
            tool: tool
        });

        results_element.style.display = "block";
        results_element.innerHTML = '<p class="line2">Fetching results...</p>';


        jQuery.ajax({
            contentType: "application/json",
            dataType: 'json',
            processData: false,
            type: 'GET',
            url: server_url,
            success: function(ajaxResult) {
                let toolData = "";

                if(tool === "ping") {
                    for (let time of ajaxResult.results) {
                        toolData = toolData + `icmp_seq=${time.seq_number} time=${time.ping_time} ms<br/>`
                    }
                   //toolData = toolData + `<br/>Ping statistics: Average: ${jsonData.avg} ms Minimum: ${jsonData.min} ms Deviation: ${jsonData.stddev} ms Packet Loss: ${jsonData.packetLoss} %<br/><br/>`;

                    if (!ajaxResult.alive) {
                        results_element.innerHTML = `<p class="line2" style="color: red;">*The host you specified is either down or invalid!</p><br/><br/>`
                    } else {
                        results_element.innerHTML = `<p class="line2">${toolData}</p>`;
                    }
                } else if(tool === "trace") {
                    if(ajaxResult.hops === 'undefined') {
                        results_element.innerHTML = `<p class="line2" style="color: red;">*The host you specified is invalid!</p><br/><br/>`;
                    } else {
                        for (let hop of ajaxResult.hops) {
                            toolData += `<p class="line2">${hop}</p>`;
                        }
                        results_element.innerHTML = `<p class="line2">${toolData}</p>`;
                    }
                }

                // Enable the button
                submit_element.disabled = false;

            },
            error: function() {
                results_element.style.display = "block";
                results_element.innerHTML = `<p id="lgResultScreen" class="line2" style = "color: red;">The host you specified isn't valid or an error<br> has occurred.<span class="cursor2">_</span></p>`;

                submit_element.disabled = false;
            }
        });

    }
