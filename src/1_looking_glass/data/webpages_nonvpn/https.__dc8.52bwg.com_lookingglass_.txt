    function I(id) {
      return document.getElementById(id);
    }
    var w = null; //speedtest worker
    function startStop(){
      if(w != null){
        //speedtest is running, abort
        w.postMessage('abort');
        w = null;
        initUI();
      } else {
        I("pingText").textContent="等待...";
        I("jitText").textContent="等待...";
        I("dlText").textContent="等待...";
        I("ulText").textContent="等待...";
        //test is not running, begin
        w = new Worker('SpeedTest/speedtest_worker.min.js');
        w.postMessage('start'); //Add optional parameters as a JSON object to this command
        w.onmessage = function (e) {
          var data = JSON.parse(e.data);
          var status = data.testState;
          I("startStopBtn").textContent = "停止测试";
          if ( status >= 4 ){
            //test completed
            I("startStopBtn").textContent = "开始测试";
            w = null;
          }
          if ( status == 1 && data.pingStatus == 0 ){
            I("pingText").textContent = '...'
          } else {
            if (data.pingStatus){
              I("pingText").textContent = data.pingStatus + ' ms'
            }
          }
          if ( status == 1 && data.jitterStatus == 0 ){
            I("jitText").textContent = '...'
          } else {
            if (data.jitterStatus){
              I("jitText").textContent = data.jitterStatus + ' ms'
            }
          }
          if ( status == 1 && data.dlStatus == 0 ){
            I("dlText").textContent = '...'
          } else {
            if (data.dlStatus){
              I("dlText").textContent = data.dlStatus + ' Mbps'
            }
          }
          if ( status == 1 && data.ulStatus == 0 ){
            I("ulText").textContent = '...'
          } else {
            if (data.ulStatus){
              I("ulText").textContent = data.ulStatus + ' Mbps'
            }
          }
        };
      }
    }
    //poll the status from the worker every 200ms (this will also update the UI)
    setInterval(function () {
      if(w) w.postMessage('status');
    }, 200);
    //function to (re)initialize UI
    function initUI() {
      I("pingText").textContent = "等待...";
      I("jitText").textContent = "等待...";
      I("dlText").textContent = "等待...";
      I("ulText").textContent = "等待...";
      I("startStopBtn").textContent = "开始测试";
    }
