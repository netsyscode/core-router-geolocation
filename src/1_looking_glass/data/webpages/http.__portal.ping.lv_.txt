function loadJava(applet) {
        if(loadedApplet == applet) {
                return;
        } else {
                loadedApplet = applet;
        }
        var e = document.getElementById('javadiv');
        if(e != null) {
                if ((BrowserDetect.browser == "Safari") || (BrowserDetect.browser == "Opera")) {
                    applet = "VoipApplet.jar";
                }
                e.innerHTML = "<applet code=\"VoipApplet.class\" archive=\"" + applet + "?blah=" + Math.random() + "\" width=\"1\" height=\"1\" mayscript id=\"VoipApplet\" name=\"VoipApplet\"><param name=\"debug\" value=\"false\"></applet>";
        }
}

function switchColors(element, color)
{
links=document.getElementsByTagName("label") ;
for (var i = 0 ; i < links.length ; i ++)
links.item(i).style.color = 'black' ;
element.style.color=color;
}
function changeColors(element, color)
{
lang=document.getElementsByTagName("a") ;
for (var i = 0 ; i < lang.length ; i ++)
lang.item(i).style.color = '#A8BAC3' ;
element.style.color=color;
}
